    private void finalizarVendaActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_finalizarVendaActionPerformed
        // TODO add your handling code here:
        DAOVenda daovenda = new DAOVenda();
        DAOItemVenda daoitem = new DAOItemVenda();
        Venda novaVenda = new Venda();

        try {
            novaVenda.setIdVenda(codigoVenda);
            novaVenda.setIdCliente(Integer.parseInt(codigoCliente.getText()));
            novaVenda.setDataVenda(new java.sql.Date(new Date().getTime()));
            novaVenda.setDescontoVenda(Double.parseDouble(descontoVenda.getText().isEmpty()?"0.00":descontoVenda.getText()));

            novaVenda.setTipoPagVenda(tipoPagamento.getSelectedItem().toString());
            novaVenda.setTotalVenda(precoTotal);
            
            daovenda.insere(novaVenda);
            
            System.out.print(listaProduto);
            
            this.listaProduto.stream().forEach((produto) -> {
                try {
                    daoitem.insere(produto);
                } catch (Exception ex) {
                    imprimeMsg("Erro ao inserir o item: "+produto.getNomeProduto()+" no banco.");
                }
            });
            imprimeMsg("Compra finalizada com Sucesso.");
            this.dispose();
        } catch (Exception ex) {
            imprimeMsg("Erro ao inserir Venda no Banco");
        }

    public static byte[] gerarHash(Date dataHora) {
        try {
            MessageDigest md = MessageDigest.getInstance("MD5");
            md.update(dataHora.toString().getBytes());
            return md.digest();
        } catch (NoSuchAlgorithmException e) {
            return null;
        }
      }
    private static String stringHexa(byte[] bytes) {
        StringBuilder s = new StringBuilder();
        for (int i = 0; i < bytes.length; i++) {
            int parteAlta = ((bytes[i] >> 4) & 0xf) << 4;
            int parteBaixa = bytes[i] & 0xf;
            if (parteAlta == 0) s.append('0');
            s.append(Integer.toHexString(parteAlta | parteBaixa));
        }
        return s.toString();
     }
    
    private void getCodigo(){
        this.codigoVenda = stringHexa(gerarHash(new Date()));
    }
    
    private ItemVenda getProduto() throws Exception{
        ItemVenda produto = new ItemVenda();
        
        produto.setIdProduto(Integer.parseInt(codigoProduto.getText()));
        produto.setIdVenda(this.codigoVenda);
        produto.setNomeProduto(nomeProduto.getText());
        produto.setPrecoProduto(Double.parseDouble(precoProduto.getText()));
        produto.setQtdProduto(Integer.parseInt(qtdProduto.getText()));
        
        return produto;
    }
                
    private Object[] getLinha(ItemVenda produto){
        return new Object[]{produto.getNomeProduto(), 
                            produto.getPrecoProduto(), 
                            produto.getQtdProduto(),
                            produto.getQtdProduto()*produto.getPrecoProduto(),                            
                };
    }
    
    private void setTotalVenda(){
        DecimalFormat df = new DecimalFormat("0.##");
        int n = tabelaProduto.getRowCount();
        Double preco = 0.0;
        
        for (int i = 0; i < n; i++)
            preco +=  Double.parseDouble(tabelaProduto.getValueAt(i, 3).toString());
        
        this.precoTotal = preco;
        
        totalVenda.setText(df.format(this.precoTotal));
    }
    
    private void setTotalPagar(){
        DecimalFormat df = new DecimalFormat("0.##");
        double desc = 0;
        
        if(descontoVenda.getText().contains("%")){
            double perc = Double.parseDouble(descontoVenda.getText().replace("%", ""));
            desc = this.precoTotal*(perc/100);
        }else if(!descontoVenda.getText().isEmpty()){
            desc = Double.parseDouble(descontoVenda.getText());
        }
        
        if(desc <= this.precoTotal)
            totalPagar.setText(df.format(this.precoTotal - desc));
        else
            imprimeMsg("Descoto maior que o total. WTF??");
    }
    
    private Produto buscarProduto(int id){        
        try {
            DAOProduto daoProduto = new DAOProduto();
